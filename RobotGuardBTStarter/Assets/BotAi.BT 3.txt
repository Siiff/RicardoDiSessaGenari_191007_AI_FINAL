//Raiz// //Persegue e atira//
tree("Root")
		parallel 
			repeat mute tree("Patrol")
			repeat mute tree("Dead")
//Vaga pelo cenario//
tree("Wander")
	sequence
		PickRandomDestination
		MoveToDestination
//Patrulha//
tree("Patrol")
		fallback 
			tree("Attack")
			while IsHealthLessThan(40.0)
				fallback
					tree("Wander")
//Ataca//
tree("Attack")
	while SeePlayer
		sequence
			TargetPlayer
			LookAtTarget
			FollowPlayer
			Wait(0.2)
			Fire

//Morte//
tree("Dead")
	sequence
		IsHealthLessThan(0.1)
		Explode



//Script de AI do bot, executa em paralelo as ações//
