//Raiz// //Ataca só se for atacado//
tree("Root")
		parallel 
			repeat mute tree("Patrol")
			repeat mute tree("Dead")
//Vaga pelo cenario//
tree("Wander")
	sequence
		PickRandomDestination
		MoveToDestination
//Patrulha//
tree("Patrol")
		fallback 
			tree("Wander")
			while IsHealthLessThan(70.0)
				fallback
					tree("Attack")
					//Ataca se tiver com a vida abaixo de X
//Ataca//
tree("Attack")
	while SeePlayer
		sequence
			TargetPlayer
			LookAtTarget
			Wait(0.2)
			Fire

//Morte//
tree("Dead")
	sequence
		IsHealthLessThan(0.1)
		Explode
